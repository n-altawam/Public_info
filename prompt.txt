import asyncio
import httpx
import os
from tqdm.asyncio import tqdm
from certara_secret import oauth2_token, cert_verify

# SSL Config
os.environ["SSL_CERT_FILE"] = cert_verify
os.environ["REQUEST_CA_BUNDLE"] = cert_verify

async def process_batch(client, semaphore, batch, url, token):
    """Sends a single batch request with concurrency control."""
    async with semaphore:
        payload = {
            "inputText": batch,
            "embedder": "multi-qa-MiniLM·l6-cos-vl"
        }
        try:
            response = await client.post(
                url, 
                json=payload, 
                headers={"Authorization": f"Bearer {token}", "Accept": "Application/json"},
                timeout=60.0
            )
            return response.json() if response.status_code == 200 else {"error": response.text}
        except Exception as e:
            return {"error": str(e)}

async def embed_all_texts(texts, batch_size=100, max_concurrency=10):
    """Splits texts into batches and processes them concurrently."""
    api_url = "YOUR_API_URL_HERE" # Replace with your get_api_service_url()
    
    # Create batches
    batches = [texts[i : i + batch_size] for i in range(0, len(texts), batch_size)]
    semaphore = asyncio.Semaphore(max_concurrency)
    
    async with httpx.AsyncClient(verify=cert_verify) as client:
        tasks = [
            process_batch(client, semaphore, b, api_url, oauth2_token) 
            for b in batches
        ]
        
        # tqdm.gather handles the progress bar for async tasks
        results = await tqdm.gather(*tasks, desc="Processing Embeddings")
        return results

# Execution
if __name__ == "__main__":
    # Example: 100k strings
    large_list_of_texts = ["sample text"] * 100000 
    results = asyncio.run(embed_all_texts(large_list_of_texts))
